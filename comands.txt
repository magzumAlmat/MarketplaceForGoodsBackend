# Select Product
GET /api/store/products/1
# This request retrieves the details of a specific product by its ID.

# Add Product to Cart
POST /cart
{
  "productId": 1,
  "quantity": 1
}
# This request adds a product to the user's cart. The product ID and quantity are specified in the request body.

# View Cart
GET /cart
# This request retrieves the current contents of the user's cart.

# Update Cart Item Quantity
PUT /cart
{
  "productId": 1,
  "quantity": 2
}
# This request updates the quantity of a specific product in the user's cart. The product ID and new quantity are specified in the request body.

# Remove Product from Cart
DELETE /cart
{
  "productId": 1
}
# This request removes a specific product from the user's cart. The product ID is specified in the request body.
